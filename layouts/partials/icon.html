{{- /* icon — inline an SVG by name. Source: vendored files in assets/icons/ (Lucide UI icons, ISC; Simple Icons brand glyphs, CC0 — see assets/icons/README.md). Brand glyphs ship without a fill, so we theme them with currentColor. A small inline map covers bespoke marks that aren't vendored. Inlined at BUILD time — no web font, no client JS, no extra requests; only referenced icons ship. */ -}} {{- $name := . -}} {{- with resources.Get (printf "icons/%s.svg" $name) -}} {{- $svg := .Content -}} {{- if not (strings.Contains $svg "currentColor") -}} {{- $svg = replaceRE "" -}} {{- with index $custom $name -}} {{- . | safeHTML -}} {{- else -}} {{- warnf "icon: unknown icon %q (no assets/icons/%s.svg and no inline fallback)" $name $name -}} {{- (index $custom "help") | safeHTML -}} {{- end -}} {{- end -}}
