{{- /* filetree — a styled folder/file tree. Author with 2 spaces per level. A leading "* " marks the highlighted "book" row. " // note" adds a muted annotation. Trailing "/" = folder; otherwise the icon is chosen from the file extension. {{< filetree >}} Audiobooks/ Brandon Sanderson/ // author Stormlight Archive/ // series * The Way of Kings/ // a book 01 - Prologue.m4b cover.jpg {{< /filetree >}} */ -}} {{- $icons := dict "folder" "" "audio" "" "image" "" "doc" "" -}}
{{- range (split (trim .Inner "\n") "\n") -}} {{- $raw := . -}} {{- if trim $raw " " -}} {{- $body := strings.TrimLeft " " $raw -}} {{- $depth := div (sub (len $raw) (len $body)) 2 -}} {{- $book := false -}} {{- if strings.HasPrefix $body "* " }}{{ $book = true }}{{ $body = strings.TrimPrefix "* " $body }}{{ end -}} {{- $note := "" -}} {{- if strings.Contains $body " // " }}{{ $p := split $body " // " }}{{ $body = index $p 0 }}{{ $note = index $p 1 }}{{ end -}} {{- $name := trim $body " " -}} {{- $type := "file" -}} {{- if strings.HasSuffix $name "/" -}} {{- $type = "folder" -}}{{- $name = strings.TrimSuffix "/" $name -}} {{- else -}} {{- $ext := lower (path.Ext $name) -}} {{- if in (slice ".m4b" ".mp3" ".m4a" ".flac" ".opus" ".ogg" ".oga" ".aac" ".wav" ".aiff") $ext }}{{ $type = "audio" }} {{- else if in (slice ".jpg" ".jpeg" ".png" ".webp") $ext }}{{ $type = "image" }} {{- else if in (slice ".json" ".opf" ".nfo" ".txt") $ext }}{{ $type = "doc" }}{{ end -}} {{- end -}} {{- $ic := "doc" -}} {{- if eq $type "folder" }}{{ $ic = "folder" }}{{ else if eq $type "audio" }}{{ $ic = "audio" }}{{ else if eq $type "image" }}{{ $ic = "image" }}{{ end -}}

{{- end -}} {{- end -}}
