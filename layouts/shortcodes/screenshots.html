{{- /* screenshots — render one or more app screenshots from assets/images/ side by side, resized + converted to WebP by Hugo's image processing. Auto-labels iOS / Android from the filename. Usage: {{< screenshots "server-select-ios.png" "server-select-android.png" >}} */ -}}
{{- range .Params -}} {{- $name := . -}} {{- with resources.Get (printf "images/%s" $name) -}} {{- $img := .Process "resize x1080 q85 webp" -}} {{- $full := .Process "resize x2000 q85 webp" -}} {{- $label := "" -}} {{- if strings.Contains (lower $name) "ios" }}{{ $label = "iOS" }}{{ else if strings.Contains (lower $name) "android" }}{{ $label = "Android" }}{{ end -}} {{- else -}} {{- warnf "screenshots shortcode: image not found at assets/images/%s" $name -}} {{- end -}} {{- end -}}
